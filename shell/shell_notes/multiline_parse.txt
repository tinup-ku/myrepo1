#!/bin/bash

# Function to parse the input
parse_input() {
    local input="$1"
    local start_pattern="==========START============"
    local end_pattern="===========END========="
    local started=false
    local content=""

    # Loop through each line of the input
    while IFS= read -r line; do
        #echo "Running now"
        #echo "Line: $line"
        if [[ $line == "$start_pattern" ]]; then
            echo "Start found"
            echo "$line"
            started=true
            content=""
        elif [[ $line == "$end_pattern" ]]; then
            if [ "$started" = true ]; then
                # Process the content
                echo -n "$content"
                echo "$line"
            fi
            started=false
        elif [ "$started" = true ]; then
            # Append line to content
            content+="$line"$'\n'
        fi
    done < "$input"
}

# Call the function to parse the input
parse_input data

exit 0
