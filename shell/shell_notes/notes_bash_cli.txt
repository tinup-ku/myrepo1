Take out anything in file1 from file2 ( matches on first work only )
$ cat reduce.sh
#!/bin/bash
# bash reduce.sh file1 file2 ( file1 is small &  file2 is big )
# remove all lines present in file1 from file2
file1=$1
file2=$2
cat $file1 | awk '{print $1}' > temp_file1
cat $file2 | awk '{print $1}' > temp_file2
comm -13 <(sort temp_file1) <(sort temp_file2)
rm temp_file1 temp_file2
